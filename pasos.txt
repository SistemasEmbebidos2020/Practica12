// Arduino / ESP32 Code for Systemd Service Configuration

/**
 * Create a systemd service to run a Python script on startup.
 *
 * @author [Your Name]
 */

#include <Arduino.h>

// Define constants
const char* PYTHON_EXE = "/usr/bin/python3";
const char* SCRIPT_PATH = "/home/pi/scripts/mi_script.py";

void setup() {
  // Open the terminal and create a directory for scripts
  Serial.println("Abre la terminal y crea un directorio llamado 'scripts' en tu carpeta de inicio:");
  Serial.println("mkdir /home/pi/scripts");

  // Create the Python script with the correct shebang line
  Serial.println("Escribe lo que quieres que se ejecute en tu script:");
  Serial.print("echo '#!/usr/bin/env python3' > ");
  Serial.print(SCRIPT_PATH);
  Serial.println("; echo 'python code here...' >> ");
  Serial.print(SCRIPT_PATH);

  // Give execution permissions to the script
  Serial.println("Dale permisos de ejecución al script:");
  Serial.print("sudo chmod +x ");
  Serial.print(SCRIPT_PATH);

  // Create a systemd service file for the Python script
  Serial.println("Los archivos de servicio del usuario se crean en /etc/systemd/system/.");
  Serial.println("Usa nano con sudo para crear el archivo:");
  Serial.print("sudo nano /etc/systemd/system/mi_app.service");

  // Copy and paste the following content into the editor, modifying the path and name of your script
  String serviceContent = "[Unit]\n";
  serviceContent += "Description=Mi Script de Python de Ejemplo\n";
  serviceContent += "After=network.target\n\n";
  serviceContent += "[Service]\n";
  serviceContent += "ExecStart=" + PYTHON_EXE + " " + SCRIPT_PATH + "\n";
  serviceContent += "Restart=on-failure\n";
  serviceContent += "RestartSec=5\n";
  serviceContent += "User=pi\n";
  serviceContent += "WorkingDirectory=/home/pi/scripts\n";
  serviceContent += "StandardOutput=journal\n";
  serviceContent += "StandardError=journal\n\n";
  serviceContent += "[Install]\n";
  serviceContent += "WantedBy=multi-user.target";

  // Save the file and exit
  Serial.println("Guarda el archivo y ciérralo (Ctrl+X, luego Y, luego Enter):");
}

void loop() {
  // Inform systemd that a new service has been added
  Serial.println("Informa a systemd que has añadido un nuevo servicio:");
  Serial.print("sudo systemctl daemon-reload");

  // Enable the service to start automatically on boot
  Serial.println("Habilita tu servicio. Esto crea un enlace simbólico que hace que se inicie automáticamente en cada arranque:");
  Serial.print("sudo systemctl enable mi_app.service");

  // Start the service manually
  Serial.println("Inicia el servicio:");
  Serial.print("sudo systemctl start mi_app.service");

  // Check the status of the service to ensure it's running without errors
  Serial.println("Verifica el estado para asegurarte de que está funcionando sin errores:");
  Serial.print("sudo systemctl status mi_app.service");
}
